00049 /* You must choose exactly one of these malloc packages:
00050  *     ~~~~
00051  * SYSMALLOC:
00052  *   * Built-in system malloc.
00053  *   * No statistics.
00054  *   * SYSMALLOC incurs no additional CPU or memory overhead.
00055  *
00056  * MALLOC64
00057  *   * Wodan's malloc, uses system malloc for small allocations and spreads
00058  *   * large allocations through the 64 bit memory space
00059  *   * won't work on 32 bit systems.
00060  * MALLOC32
00061  *   * fixes realloc by always doing a malloc/memcpy/free instead, try this
00062  *   * if you use more memory than expected (or MALLOC64 on a 64bit system).
00063  */
00064 #define SYSMALLOC
00065 #undef MMALLOC
00066 #undef MALLOC64
00067 #undef MALLOC32
00068 #undef GCMALLOC /* needs -lgc in system_libs */
00069 /* You may optionally choose one (or none) of these malloc wrappers.  These
00070  * can be used in conjunction with any of the above malloc packages.
00071  *
00072  * WRAPPEDMALLOC:
00073  *   * Limited statistics.
00074  *   * Limited additional cpu overhead and no additional memory overhead.
00075  *
00076  * DEBUGMALLOC:
00077  *   * Statistics on precisely how much memory has been malloc'd (as well
00078  *     as the stats provided by WRAPPEDMALLOC).
00079  *   * Incurs a fair amount of overhead (both memory and CPU)
00080  */
00081 #undef WRAPPEDMALLOC
00082 #undef DEBUGMALLOC

00151 /* explode():
00152  *
00153  * The old behavior (#undef both of the below) strips any number of
00154  * delimiters at the start of the string, and one at the end.  So
00155  * explode("..x.y..z..", ".") gives ({ "x", "y", "", "z", "" })
00156  *
00157  * SANE_EXPLODE_STRING strips off at most one leading delimiter, and
00158  * still strips off one at the end, so the example above gives
00159  * ({ "", "x", "y", "", "z", "" }).
00160  *
00161  * REVERSIBLE_EXPLODE_STRING overrides SANE_EXPLODE_STRING, and makes
00162  * it so that implode(explode(x, y), y) is always x; i.e. no delimiters
00163  * are ever stripped.  So the example above gives
00164  * ({ "", "", "x", "y", "", "z", "", "" }).
