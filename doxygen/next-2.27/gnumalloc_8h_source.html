00021 /* The allocator divides the heap into blocks of fixed size; large
00022    requests receive one or more whole blocks, and small requests
00023    receive a fragment of a block.  Fragment sizes are powers of two,
00024    and all fragments of a block are the same size.  When all the
00025    fragments in a block have been freed, the block itself is freed.
00026    WARNING: BLOCKSIZE must be set greater than or equal to the
00027    machine's page size for valloc() to work correctly.  The default
00028    definition here is 4096 bytes. */
00029 #define INT_BIT (CHAR_BIT * sizeof (int))
00030 #define BLOCKLOG (INT_BIT > 16 ? 12 : 9)
00031 #define BLOCKSIZE (1 << BLOCKLOG)
00032 #define BLOCKIFY(SIZE) (((SIZE) + BLOCKSIZE - 1) / BLOCKSIZE)
00033

00042 /* Data structure giving per-block information. */
00043 union info {
00044     struct {

00046                    giving the logarithm to the base two
00047                    of the fragment size. */
00048     union {
00049         struct {
00050         int nfree;  /* Free fragments in a fragmented block. */
00051         int first;  /* First free fragment of the block. */
00052         } frag;
00053         int size;       /* Size (in blocks) of a large cluster. */
00054     } info;
00055     } busy;
00056     struct {
00057     int size;       /* Size (in blocks) of a free cluster. */
